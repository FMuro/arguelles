      \begin{proof}
        \begin{equation*}
          e^{iz}=\cos{z}+i\sin{z}
        \end{equation*}
        \center{therefore}
        \begin{align*}
          e^{i\pi}+1 & \null=\cos\pi+i\sin\pi+1 \\
          & \null=-1+i\times0+1 \\
          & \null=0
        \end{align*}
      \end{proof}
    \end{theorem}
